python多线程

May 25, 2020点击

每次用到python多线程,都是看了就忘

threading

用法:threading.Thread(target=函数名)创建线程,start开始执行,join等待完成

示例:

import threading

reader_count = 0
write_count = 0
reader_amount = 0
write_amount = 0


def reader():
    global reader_count, write_count, reader_amount, write_amount
    while reader_amount > 0:
        if reader_count > 0 or write_count == 0:
            reader_count += 1
            print('read')
            reader_amount -= 1
            reader_count -= 1
        else:
            print("can't read")


def writer():
    global reader_count, write_count, reader_amount, write_amount
    while write_amount > 0:
        if reader_count > 0 or write_count > 0:
            print("can't write")
        else:
            write_count += 1
            print('write')
            write_amount -= 1
            write_count -= 1


if __name__ == '__main__':

    reader_amount = int(input('读次数:'))
    write_amount = int(input('写次数:'))
    read = threading.Thread(target=reader)
    write = threading.Thread(target=writer)
    write.start()
    read.start()
