马拉车算法

刷leetcode时,遇到了这道题

粗一分析,发现本质上是在找这个字符串的最长回文前缀,将非前缀的转置加到前面就是结果了

超时代码

class Solution:
    def checkP(self, s: str):
        for i in range(int(len(s) / 2)):
            if s[i] != s[len(s) - 1 - i]:
                return False
        return True

    def shortestPalindrome(self, s: str) -> str:
        if self.checkP(s):
            return s
        # t = ''
        # for i in range(len(s))[::-1]:
        #     t = t + s[i]
        #     p = t + s
        #     if self.checkP(p):
        #         return p
        n=len(s)
        max_num=0
        for i in range(2,n):
            if self.checkP(s[:i]):
                max_num=max(max_num,i)
        if max_num==0:
            return s[1:][::-1]+s
        return s[max_num:][::-1]+s

每个中心遍历一遍,重复判断回文

时间复杂度O(n^2)

马拉车

class Solution:
    def checkP(self, s: str):
        for i in range(int(len(s) / 2)):
            if s[i] != s[len(s) - 1 - i]:
                return False
        return True

    def shortestPalindrome(self, s: str) -> str:
        def manacher(s1: str):
            s1 = "#" + "#".join(s1) + "#"
            n = len(s1)
            p = [0] * n
            center, max_right = 0, 0
            for i in range(n):
                length = 0
                if i < max_right:
                    length = min(p[center * 2 - i], max_right - i)
                while i - length >= 0 and i + length < n and s1[i + length] == s1[i - length]:
                    if length + i > max_right:
                        max_right = length + i
                        center = i
                    length += 1
                p[i] = length - 1
            return p

        if self.checkP(s):
            return s
        n = len(s)
        p = manacher(s)
        max_num = 0
        for i in range(len(p)):
            if i - p[i] == 0:
                max_num = max(max_num, (i + p[i])//2)
        if max_num == 0:
            return s[1:][::-1] + s
        return s[max_num:][::-1] + s

时间复杂度O(n)
